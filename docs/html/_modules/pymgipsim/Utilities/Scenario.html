    simulator_name: str
    solver_name: str

    """

    input_generation: input_generation
    inputs: inputs
    patient: patient
    controller: controller

def load_scenario(path):

    with open(path, "r") as f: #
        loaded_scenario = json.load(f)
    f.close()

    scenario_instance = scenario(**loaded_scenario)

    return scenario_instance
