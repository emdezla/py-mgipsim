    def preprocessing(self):

        """
        This is going to function similar to 'initialize' the way I had it before. output_equilibrium does the initial condition
        calculation so this will just call that static method for this model.

        The values in self.initial_conditions can be assigned individually based on that the output_equilibrium (e.g. self.initial_conditions
        .Sg = 0)
        """
        self.initial_conditions.as_array = self.output_equilibrium(self.parameters.as_array, self.inputs.as_array)

        self.states.as_array = np.zeros((self.parameters.as_array.shape[0], self.initial_conditions.as_array.shape[1], self.time.as_unix.size))





    @staticmethod
    @njit(cache = True)
    def model(states, time, parameters, inputs) -> np.ndarray:
        
        # States 
        Sg, Cp, Ga, Gt, Gc, Gp, GE, I, IE = states.T
        
        # Parameters
        BW0, RTG0, GFR0, RTG_Max, EC50, ksa, ke, Vd, ka, Gprod0, Q, Vp, Vg, ClG, ClGI, kGE, GSS, ISS, IPRG, Sincr, Isec0, ClI, Vi, kIE = parameters.T
        
        # Inputs
        uFastCarbs, uSlowCarbs, uSGLT2I, uBWE = inputs.T

        uCarb = (uFastCarbs + uSlowCarbs) * 1000 / 180.156

        uSGLT2I *= 1e6

        """ Canagliflozin equations """
        dSgdt = uSGLT2I - ksa*Sg
        
        dCpdt = ksa*Sg / Vd - ke*Cp
            
        RTG = RTG0*(1 - RTG_Max*Cp/(Cp+EC50))
        
        
        """ Renal Equations """
        RC = RTG*GFR0
        
        GluFR = GFR0 * (Gc/Vg)
            
        Rreab = np.minimum(GluFR, RC)
        
        RUGE = GluFR - Rreab
        
        
        """ Glucose Equations """
        # Ga
        dGadt = uCarb - ka*Ga
        
        # Gt
        dGtdt = ka * (Ga - Gt)
    
        # Central compartment
        dGcdt = 0.78*ka*Gt + Gprod0 + (Q/Vp)*Gp - (ClG + ClGI*IE + Q) * Gc / Vg - RUGE
        
        # Peripheral compartment
        dGpdt = Q * (Gc/Vg - Gp/Vp)
        
        # Effect 1
        dGEdt = kGE*Gc/Vg - kGE*GE
                    
        GCM   = (GE / GSS)**IPRG
            
        
        """ Insulin Equations """
        # Secretion
        Iabsg = 1 + Sincr*Gt
        
        Isec = Isec0 * GCM * Iabsg
    
        # Insulin
        dIdt = Isec - (ClI/Vi)*I
        
        # Effect
        dIEdt = kIE * (I / (Vi * uBWE)  - IE)
                
        return np.column_stack((dSgdt, dCpdt, dGadt, dGtdt, dGcdt, dGpdt, dGEdt, dIdt, dIEdt))

    @staticmethod
    def rate_equations(states, time, parameters, inputs):
        
        Sg, Cp, Ga, Gt, Gc, Gp, GE, I, IE = states.swapaxes(0, 1)
        
        # Parameters
        BW0, RTG0, GFR0, RTG_Max, EC50, ksa, ke, Vd, ka, Gprod0, Q, Vp, Vg, ClG, ClGI, kGE, GSS, ISS, IPRG, Sincr, Isec0, ClI, Vi, kIE = parameters.T
                            
        GCM   = np.power(np.divide(GE, GSS[:, None]), IPRG[:, None])

        Iabsg = 1 + Sincr[:, None]*Gt
        
        Isec = Isec0[:, None] * GCM * Iabsg
        
        """ Renal Equations """
        
        RTG = RTG0[:, None] * (1 - RTG_Max[:, None]*Cp/(Cp+EC50[:, None]))
        
        RC = RTG*GFR0[:, None]
        
        GluFR = GFR0[:, None] * (Gc/Vg[:, None])
        
        Rreab = np.minimum(RC, GluFR)
        
        RUGE = GluFR - Rreab


        return np.stack((GCM, Iabsg, Isec, RTG, RC, GluFR, Rreab, RUGE))

    @staticmethod
    def output_equilibrium(parameters: np.ndarray, inputs: np.ndarray) -> np.ndarray:

        """
        This calculates the steady state values for each state variable
        """
        BW0, RTG0, GFR0, RTG_Max, EC50, ksa, ke, Vd, ka, Gprod0, Q, Vp, Vg, ClG, ClGI, kGE, GSS, ISS, IPRG, Sincr, Isec0, ClI, Vi, kIE = parameters.T

        N = parameters.shape[0]

        Sg0 = np.zeros(N)
        Cp0 = np.zeros(N)
        Ga0 = np.zeros(N)
        Gt0 = np.zeros(N)
        Gc0 = GSS * Vg
        Gp0 = GSS * Vp
        GE0 = GSS
        I0  = ISS * Vi
        IE0 = ISS

        return np.column_stack((Sg0, Cp0, Ga0, Gt0, Gc0, Gp0, GE0, I0, IE0))



    @staticmethod
    def from_scenario(scenario_instance: scenario):
        time = np.arange(scenario_instance.settings.start_time,
                         scenario_instance.settings.end_time,
                         scenario_instance.settings.sampling_time
                         )

        model = Model(sampling_time=scenario_instance.settings.sampling_time)

        events = scenario_instance.inputs.meal_carb
        meal_carb = Signal(time=time, sampling_time=scenario_instance.settings.sampling_time, duration=events.duration,
                           start_time=events.start_time, magnitude=events.magnitude)
        
        events = scenario_instance.inputs.snack_carb
        snack_carb = Signal(time=time, sampling_time=scenario_instance.settings.sampling_time, duration=events.duration,
                            start_time=events.start_time, magnitude=events.magnitude)
        
        events = scenario_instance.inputs.sgl2i
        sglt2i = Signal(time=time, sampling_time=scenario_instance.settings.sampling_time, duration=events.duration,
                        start_time=events.start_time, magnitude=events.magnitude)
        
        events = scenario_instance.inputs.bodyweighteffect
        bodyweighteffect = Signal(time=time, sampling_time=scenario_instance.settings.sampling_time, duration=events.duration,
                                start_time=events.start_time, magnitude=events.magnitude)
        

        model.inputs = Inputs(sglt2i=sglt2i, slow_carbs=meal_carb, fast_carbs=snack_carb, bodyweighteffect = bodyweighteffect)
        model.parameters = Parameters(np.asarray(scenario_instance.patient.model.parameters))
        model.time = Timestamp()
        model.time.as_unix = time

        return model
