class Controller:
    name = "mvAP"
    def __init__(self, scenario_instance: scenario, initital_glucose):
        demographic_info = scenario_instance.patient.demographic_info
        self.basal_vector = UnitConversion.insulin.Uhr_to_mUmin(np.asarray(demographic_info.basal)).tolist() * 4  # [1, 1, 1, 1] * virtual_patient.demo_info.get_basal_rate_mU_min()
        self.bolus_vector = [0] * 4  # [0, 0, 0, 0]
        self.fast_carb_vector = [0] * 4

        self.sampling_time = scenario_instance.settings.sampling_time

        self.demographic_info = Demographic_Information(BW=demographic_info.body_weight[0], height=demographic_info.height[0], insulin2carb_ratio=demographic_info.carb_insulin_ratio[0],
                                                        correction_factor=demographic_info.correction_bolus[0], basal_rate=demographic_info.basal[0],
                                                        HbA1c=demographic_info.HbA1c[0], waist_size=demographic_info.waist_size[0])

        # sample time
        self.delta = 5

        self.measurement_counter = 0

        self.counter = 0

        num_observer_states = 13
        n_Ug = 10
        model_NRLS = 2

        self.noise_error_cov_mat = Noise_Error_Covariance_Matrix()
        self.noise_error_cov_mat.P_UKF = 0.1 * np.eye(num_observer_states)
        self.noise_error_cov_mat.Q_UKF = np.diag(
            [1e-10, 1e-10, 1e-10, 1e-10, 1e-10, 2.668e-08, 1e-10, 1e-10, 1e-10, 1e-10, 4.65612656105701, 1e-10,
             0.138709042564221])
        self.noise_error_cov_mat.R_UKF = 0.4
        self.noise_error_cov_mat.P_NRLS = 0.01 * np.eye(2)

        self.pw_data = Data_PW()
        self.pw_data.cgm_pw.extend(initital_glucose.tolist() * 36)
        self.pw_data.push_basal(self.demographic_info.basal_rate)
        self.pw_data.push_gut_absorption_rate(0)

        self.hovorka_params = Hovorka_Parameters()
        self.hovorka_params.BW = self.demographic_info.BW

        # finding steady_state condition as the initial condition
        xss = Hovorka_Model.get_hovorka_model_steady_state_vector(self.hovorka_params, self.pw_data.get_last_cgm(),
                                                                  self.pw_data.get_last_basal_mU_min())
        self.X = Hovorka_Model_Extended_States_Matrix()
        self.X.push_current_state_vector(xss)

        self.gut_absorption_rate_predictor = ar_gut_absorption_rate_predictor()
        self.gut_absorption_rate_predictor.P_Ug = 100 * np.eye(n_Ug)

        self.observer_convergence_sample_time = 18
        pass

    def run(self, measurements, inputs, sample):

        if sample % self.delta==0:
            G = UnitConversion.glucose.concentration_mmolL_to_mgdL(measurements[0])

            uFastCarbs, uSlowCarbs, uHR, uInsulin, energy_expenditure = inputs[0]

            energy_expenditure = energy_expenditure[0]

            # pushing the currently measured cgm
            self.pw_data.push_cgm(G)

            # main algorithm
            cgm = self.pw_data.get_last_cgm()
            G = cgm / 18.0
            insulin_mU_min = self.pw_data.get_last_insulin_input_mU_min()
            gut_absorption_rate = self.pw_data.get_last_gut_absorption_rate()
            input_vector = np.array([insulin_mU_min, gut_absorption_rate, energy_expenditure])

            self.pw_data.push_energy_expenditure(np.mean(energy_expenditure))

            current_states = UKF_Observer.ukf_observer(self.X.get_current_state_vector(), G,
                                                       input_vector, self.noise_error_cov_mat,
                                                       self.hovorka_params)

            self.X.push_current_state_vector(current_states)

            # creating a state-object for estimated states
            current_states_object = Hovorka_Model_Extended_States_Object(current_states)

            # updating hovorka parameters
            self.hovorka_params.TmaxI = current_states_object.TmaxI
            self.hovorka_params.Ke = current_states_object.Ke

            # pushing estimated gut_absorption_rate
            self.pw_data.push_gut_absorption_rate(current_states_object.Ug)

            # update parameters by recursive least square
            Model_Identification.nonlinear_recursive_least_square(self.pw_data, self.X, self.hovorka_params, self.noise_error_cov_mat)

            # calculating the linear model
            linear_model = Hovorka_Model.get_hovorka_linear_model(current_states_object, input_vector,
                                                                  self.pw_data.get_last_cgm(), self.hovorka_params)

            # calling mpc_execute
            basal_out, bolus_out = MPC_Controller.mpc_execute(linear_model=linear_model, pw_data_object=self.pw_data,
                                                              hovorka_params=self.hovorka_params,
                                                              demo_info=self.demographic_info,
                                                              model_states=current_states_object.get_short_state_object(),
                                                              noise_error_cov_mat=self.noise_error_cov_mat,
                                                              gut_absorption_rate_predictor=self.gut_absorption_rate_predictor)

            if sample >  self.observer_convergence_sample_time:
                # basal_out, bolus_out, fast_carb_out = MPC_Controller.safety_constraints(pw_data, basal_out, bolus_out)
                fast_carb_out = 0
            else:
                basal_out = self.demographic_info.get_basal_rate()
                bolus_out = 0
                fast_carb_out = 0

            self.pw_data.push_basal(basal_out)
            self.pw_data.push_bolus(bolus_out)
            self.pw_data.push_fast_carb(fast_carb_out)

            self.basal_vector.extend([basal_out] * 5)
            self.bolus_vector.extend([bolus_out] + [0] * 4)
            self.fast_carb_vector.extend([fast_carb_out] + [0] * 4)

            self.counter = self.counter + 1
            insulin_rate = UnitConversion.insulin.Uhr_to_mUmin(self.demographic_info.basal_rate)+UnitConversion.insulin.U_to_mU(bolus_out)/self.delta
            inputs[0,3,sample:sample+self.delta] = insulin_rate
        return
