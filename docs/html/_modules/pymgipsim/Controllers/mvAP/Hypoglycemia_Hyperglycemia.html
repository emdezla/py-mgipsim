def get_hyperglycemia_probability(cgm, predicted_cgm):
    # This is optional, you can make the CGM signal smoother by doing this
    cgm_con = np.convolve(cgm, np.ones(3)/3, mode='valid')
    cgm_con = np.insert(cgm_con, 0, (cgm[0] + cgm[1]) / 2)
    cgm_con = np.append(cgm_con, (cgm[-1] + cgm[-2]) / 2)
    cgm = cgm_con

    delt = 5
    # Calculating the first derivative of CGM at the last 2 sample times
    cgm_fd = [(cgm[-1]-2*cgm[-2]+cgm[-3])/(2*delt), (cgm[-3]-4*cgm[-2]+3*cgm[-1])/(2*delt)]
    # Calculating the CGM second derivative at the last sample time
    cgm_sd = (cgm_fd[-1] - cgm_fd[-2]) / delt

    prob_cgm_trend = get_hyper_prob_cgm_trend([cgm[-1], cgm_fd[-1], cgm_sd])
    prob_cgm_prediction = get_hyper_prob_cgm_prediction(np.max(predicted_cgm))

    prob = get_hyper_prob([prob_cgm_trend, prob_cgm_prediction])

    return prob

def fuzzify_input(input_val, fuzzy_sets):
    """
    number of fuzzy membership functions is equal to the number of Fuzzy_sets entries. For example when Fuzzy_sets contain 3 elements, this means there
    are 3 fuzzy membership functions consisting of 2 trapezoid and 1 triangle. On the other hand when number of elements in Fuzzy_sets is 5, it means there are
    5 memebership functions consisting of 2 trapezoid and 3 triangles each entery of Fuzzy_sets is the point at which the fuzzy membership function is max
    """

    num_fuzzy_sets = len(fuzzy_sets)
    fuzzified_input = np.zeros(num_fuzzy_sets)

    if np.any(fuzzy_sets > input_val):
        index = np.argmax(fuzzy_sets > input_val)
    else:
        index = -1


    if index == 0:
        fuzzified_input[0] = 1
    elif index == -1:
        fuzzified_input[-1] = 1
    else:
        fuzzified_input[index-1] = (input_val - fuzzy_sets[index]) / (fuzzy_sets[index-1] - fuzzy_sets[index])
        fuzzified_input[index] = (input_val - fuzzy_sets[index-1]) / (fuzzy_sets[index] - fuzzy_sets[index-1])

    return fuzzified_input

def fuzzy_inference(rule_list, *args):
    num_output_Fuzzy_sets = int(np.max(rule_list[:, -1]))
    fuzzified_output = np.zeros(num_output_Fuzzy_sets)

    num_inputs = len(args)

    num_rules = len(rule_list)
    rule_list_value = np.zeros(num_rules)

    for i in range(num_rules):
        input_val = np.zeros(num_inputs)
        for j in range(num_inputs):
            if rule_list[i, j] != 0:
                input_val[j] = args[j][rule_list[i, j] - 1]
            else:
                input_val[j] = 1

        rule_list_value[i] = np.min(input_val)

    for i in range(1, num_output_Fuzzy_sets + 1):
        index = rule_list[:, -1] == i
        if np.any(index):
            fuzzified_output[i - 1] = np.max(rule_list_value[index])

    return fuzzified_output

def defuzzify_output(fuzzified_Output, Fuzzy_sets):
    num_output_Fuzzy_sets = len(fuzzified_Output)

    Area = np.zeros(num_output_Fuzzy_sets)
    Center = np.zeros(num_output_Fuzzy_sets)

    Area[0] = get_trapezoid_fmf_area(np.array([0, Fuzzy_sets[0], Fuzzy_sets[1]]), fuzzified_Output[0], "left")
    Center[0] = get_trapezoid_fmf_center(np.array([0, Fuzzy_sets[0], Fuzzy_sets[1]]), "left")

    for i in range(1, num_output_Fuzzy_sets - 1):
        Area[i] = get_triangle_fmf_area(np.array([Fuzzy_sets[i-1], Fuzzy_sets[i], Fuzzy_sets[i+1]]), fuzzified_Output[i])
        Center[i] = get_triangle_fmf_center(np.array([Fuzzy_sets[i-1], Fuzzy_sets[i], Fuzzy_sets[i+1]]))

    Area[-1] = get_trapezoid_fmf_area(np.array([Fuzzy_sets[-2], Fuzzy_sets[-1], 1]), fuzzified_Output[-1], "right")
    Center[-1] = get_trapezoid_fmf_center(np.array([Fuzzy_sets[-2], Fuzzy_sets[-1], 1]), "right")

    defuzzified_output = np.sum(Area * Center) / np.sum(Area)

    return defuzzified_output

def get_triangle_fmf_area(Fuzzy_set, membership_val):
    triangle_base = Fuzzy_set[2] - Fuzzy_set[0]
    triangle_area = 1/2 * triangle_base  # height is 1

    top_triangle_height = 1 - membership_val
    top_triangle_base = triangle_base * top_triangle_height
    top_triangle_area = 1/2 * top_triangle_base * top_triangle_height

    area = triangle_area - top_triangle_area

    return area

def get_trapezoid_fmf_area(Fuzzy_set, membership_val, LR):
    # LR determines the trapezoid is a left trapezoid or right trapezoid
    trapezoid_base1 = Fuzzy_set[2] - Fuzzy_set[0]

    if LR == "left":
        trapezoid_base2 = (Fuzzy_set[1] - Fuzzy_set[0]) + (1 - membership_val) * (Fuzzy_set[2] - Fuzzy_set[1])
    else:  # LR="right"
        trapezoid_base2 = (Fuzzy_set[2] - Fuzzy_set[1]) + (1 - membership_val) * (Fuzzy_set[1] - Fuzzy_set[0])

    area = 1/2 * (trapezoid_base1 + trapezoid_base2) * membership_val

    return area

def get_triangle_fmf_center(Fuzzy_set):
    centroid = 1/3 * (Fuzzy_set[0] + Fuzzy_set[1] + Fuzzy_set[2])

def get_trapezoid_fmf_center(Fuzzy_set, LR):
    # LR determines the trapezoid is a left trapezoid or right trapezoid

    if LR == "left":
        rectangle_area = Fuzzy_set[1] - Fuzzy_set[0]
        rectangle_center = 1/2 * (Fuzzy_set[0] + Fuzzy_set[1])
        triangle_area = 1/2 * (Fuzzy_set[2] - Fuzzy_set[1])
        triangle_center = Fuzzy_set[1] + 1/3 * (Fuzzy_set[2] - Fuzzy_set[1])
    else:  # LR="right"
        rectangle_area = Fuzzy_set[2] - Fuzzy_set[1]
        rectangle_center = 1/2 * (Fuzzy_set[1] + Fuzzy_set[2])
        triangle_area = 1/2 * (Fuzzy_set[1] - Fuzzy_set[0])
        triangle_center = Fuzzy_set[1] - 1/3 * (Fuzzy_set[1] - Fuzzy_set[0])

    centroid = (rectangle_area * rectangle_center + triangle_area * triangle_center) / (rectangle_area + triangle_area)

    return centroid
