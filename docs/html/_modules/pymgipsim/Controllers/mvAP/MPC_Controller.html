def safety_constraints(pw_data_object: Data_PW, basal_out, bolus_out):
    # pw_data_object contains all the past window data (usually 36 last sample times)
    # basal_out and bolus_out are basal and bolus calculated by the MPC controller

    fast_carb_out = 0  # gr

    if pw_data_object.hypo_prob_pw[-1] > 0.65 or pw_data_object.energy_expenditure_pw[-1] > 2.0:
        basal_out = 0.0  # U/hour
        bolus_out = 0.0  # U
        if np.sum(pw_data_object.fast_carb_pw) < 20.0:
            fast_carb_out = 5.0  # gr

    return basal_out, bolus_out, fast_carb_out

def get_max_bolus(pw_data_object: Data_PW, demo_info: Demographic_Information, current_PIC, setpoint_mg_dl, T=5):
    # T is sample time
    cgm_pw = np.array(pw_data_object.cgm_pw)
    bolus_pw = np.array(pw_data_object.bolus_pw)

    insulin_pump_inc = 0.05

    BW = demo_info.BW

    # cgm_max = np.max(cgm_pw[-6:])  # the max value of the CGM in the past half hour
    cgm_max = pw_data_object.get_cgm_max(6)  # the max value of the CGM in the past half hour

    # dcgm_dt = (3.0 * cgm_pw[-1] - 4.0 * cgm_pw[-2] + cgm_pw[-3]) / (2.0 * T)
    dcgm_dt = pw_data_object.get_dcgm_dt()
    # d2cgm_dt2 = (cgm_pw[-1] - 2.0 * cgm_pw[-2] + cgm_pw[-3]) / (T ** 2)
    d2cgm_dt2 = pw_data_object.get_d2cgm_dt2()
    # d2cgm_dt2 = (2.0 * cgm_pw[-1] - 5.0 * cgm_pw[-2] + 4.0 * cgm_pw[-3] - cgm_pw[-4]) / (T ** 3)

    dcgm_dt_thld = 5.0

    norm_dcgm_dt = 0
    if cgm_pw[-1] > 110.0 and dcgm_dt >= dcgm_dt_thld:
        if d2cgm_dt2 >= 0:
            norm_dcgm_dt = dcgm_dt / 20.0
        else:
            norm_dcgm_dt = dcgm_dt / 40.0

    if cgm_max > 250.0:
        max_carb = 150.0  # gram of CHO
    elif 200.0 < cgm_max <= 250.0:
        max_carb = 125.0  # gram of CHO
    else:
        max_carb = 100.0  # gram of CHO

    if np.sum(bolus_pw[-12:]) >= (max_carb / demo_info.insulin2carb_ratio) or dcgm_dt < 0:
        max_bolus = 0.001  # U
    else:
        max_bolus_rem = (max_carb / demo_info.insulin2carb_ratio) - np.sum(bolus_pw[-12:])
        max_bolus = (1 + norm_dcgm_dt) * (cgm_max - setpoint_mg_dl) / demo_info.correction_factor - current_PIC * BW * 0.12 / 1000.0  # U
        max_bolus = np.min((max_bolus, max_bolus_rem))

    max_bolus = np.round(max_bolus / insulin_pump_inc) * insulin_pump_inc
    max_bolus = max(0.001, max_bolus)

    return max_bolus

def get_glycemic_risk_index(cgm, exercise_adjustment):
    # cgm: current value of cgm in mg/dl
    # exercise_adjustment: adjustment of the lower-bound and upper-bound for the hypo and hyper
    hypo_thld = 50.0 + exercise_adjustment
    hyper_thld = 300.0 + exercise_adjustment

    safe_range_lower = 80.0 + exercise_adjustment
    safe_range_upper = 140.0 + exercise_adjustment

    risk_index = 0.0
    if cgm < hypo_thld:
        risk_index = 1.0
    elif hypo_thld <= cgm < safe_range_lower:
        risk_index = (1.0567 * (safe_range_lower - cgm) ** 1.3378) / 100.0
    elif safe_range_lower <= cgm < safe_range_upper:
        risk_index = 0.0
    elif safe_range_upper <= cgm < hyper_thld:
        risk_index = (0.4607 * (cgm - safe_range_upper) ** 1.0601) / 100.0
    elif cgm >= hyper_thld:
        risk_index = 1.0

    return risk_index

def get_exercise_adjustment(energy_expenditure_pw):
    energy_expenditure_vector = np.array(energy_expenditure_pw)
    if any(energy_expenditure_vector > 3.5):
        exercise_adj = 50
    else:
        exercise_adj = 0

    return exercise_adj

def mpc_execute(linear_model: Linear_State_Space_Model, pw_data_object: Data_PW, hovorka_params: Hovorka_Parameters,
                demo_info: Demographic_Information, model_states: Hovorka_Model_States_Object_1,
                noise_error_cov_mat: Noise_Error_Covariance_Matrix, gut_absorption_rate_predictor: ar_gut_absorption_rate_predictor):
    prediction_horizon = 24
    R0 = 1
    Q0 = 10000
    n = linear_model.get_state_num()
    m = linear_model.get_input_num_artificial()
    p = linear_model.get_output_num()
    T = linear_model.T

    # set it to either 'zero_min' or 'basal_min'
    insulin_bound_mode = 'basal_min'

    # current_basal = pw_data_object.get_basal_pw_mU_min()[-1]  # mU/min
    current_cgm = pw_data_object.cgm_pw[-1]

    ysp_mg_dl = 110.0  # mg/dl
    ysp_deviation_mmol_l = (ysp_mg_dl - current_cgm) / 18.0  # mmol/l

    exercise_adj = get_exercise_adjustment(pw_data_object.energy_expenditure_pw)
    risk_index = get_glycemic_risk_index(current_cgm, exercise_adj)

    PIC_basal = demo_info.get_basal_rate_mU_min() / (hovorka_params.get_VI() * hovorka_params.Ke)
    gamma = model_states.PIC / PIC_basal

    PIC_constraints = Plasma_Insulin_Concentration_Constraint_Params(PIC_basal)
    bounds = Bounds()

    Q_x = np.zeros((n, n))
    Q_y = Q0 * risk_index
    Q_u = np.diag([R0 * (gamma ** 2), 0, 0, 0])
    Q_du = np.zeros((m, m))
    Q_PIC = np.diag([1e5, 1e5])
    PIC_pos = model_states.get_PIC_pos()
    mpc_obj_fcn_params = MPC_Objective_Function_Params(prediction_horizon=prediction_horizon, Q_x=Q_x, Q_e=Q_y, Q_u=Q_u, Q_du=Q_du, Q_PIC=Q_PIC, PIC_pos=PIC_pos, ysp=ysp_deviation_mmol_l)

    # cgm_min = np.min(np.array(pw_data_object.cgm_pw)[-6:])  # min value of cgm in the last half hour

    max_bolus = get_max_bolus(pw_data_object, demo_info, model_states.PIC, ysp_mg_dl)

    # predict gut absorption on the prediction horizon
    gut_absorption_rate_predictor.update(pw_data_object.gut_absorption_rate_pw[-1])
    predicted_gut_absorption_rate = gut_absorption_rate_predictor.predict_filter()

    # setting bounds
    # cgm_min is the min value of cgm in the last half hour
    bounds.x_min = np.full(n, -np.inf)
    bounds.x_max = np.full(n, np.inf)
    bounds.y_min = np.full(p, -np.inf)
    bounds.y_max = np.full(p, np.inf)
    bounds.du_min = np.full(m, -np.inf)
    bounds.du_max = np.full(m, np.inf)

    bounds.u_min = np.zeros((m, prediction_horizon))
    bounds.u_max = np.zeros((m, prediction_horizon))

    if insulin_bound_mode == 'basal_min':
        if pw_data_object.get_dcgm_dt() < 0 and pw_data_object.get_cgm_min(6) < 180.0:
            bounds.u_min[0, :] = (0-pw_data_object.get_last_insulin_input_mU_min())
            bounds.u_max[0, :] = max_bolus * 1000.0 / T - pw_data_object.get_last_insulin_input_mU_min()
        else:
            bounds.u_min[0, :] = (pw_data_object.get_last_basal_mU_min() - pw_data_object.get_last_insulin_input_mU_min())
            bounds.u_max[0, :] = max_bolus * 1000.0 / T + demo_info.get_basal_rate_mU_min() - pw_data_object.get_last_insulin_input_mU_min()
    else:
        bounds.u_min[0, :] = (0 - pw_data_object.get_last_insulin_input_mU_min())
        bounds.u_max[0, :] = max_bolus * 1000.0 / T - pw_data_object.get_last_insulin_input_mU_min()

    bounds.u_min[1, :] = bounds.u_max[1, :] = predicted_gut_absorption_rate - pw_data_object.gut_absorption_rate_pw[-2]
    bounds.u_min[2, :] = bounds.u_max[2, :] = pw_data_object.energy_expenditure_pw[-1] - pw_data_object.energy_expenditure_pw[-2]
    bounds.u_min[3, :] = bounds.u_max[3, :] = 1

    qop = mpc_set(linear_model=linear_model, mpc_obj_fcn_params=mpc_obj_fcn_params, PIC_constraints=PIC_constraints, bounds=bounds)
    x_solution, exit_flag = mpc_solve(qop)

    # check if the qpsolver exit flag is successful, if not continue with the last calculated basal
    if x_solution is not None:
        sample_time_decision_var_num = m + p + p + 6 + n + m
        u_opt = x_solution.reshape((prediction_horizon, sample_time_decision_var_num))[:, -4:-1].T
        u_horizon = u_opt + np.array([
            pw_data_object.get_last_insulin_input_mU_min(),
            pw_data_object.gut_absorption_rate_pw[-2],
            pw_data_object.energy_expenditure_pw[-2]
        ]).reshape(-1, 1)
        opt_insulin_trajectory = u_horizon[0, :]
    else:
        u_horizon = np.concatenate([
            [[pw_data_object.get_last_basal_mU_min()] * prediction_horizon],
            [[pw_data_object.get_last_gut_absorption_rate()] * prediction_horizon],
            [[pw_data_object.get_last_energy_expenditure()] * prediction_horizon]
        ], axis=0)
        opt_insulin_trajectory = u_horizon[0, :]

    hypo_hyper_prediction_horizon = 12

    # u_opt is the optimal set of output insulin calculated by the mpc controller
    if pw_data_object.push_num > 18:
        predicted_cgm, _ = Hovorka_Model.get_hovorka_model_nonlinear_propagation(model_states.get_state_vector(), u_horizon, hovorka_params, hypo_hyper_prediction_horizon)

        hypo_prob = Hypoglycemia_Hyperglycemia.get_hypoglycemia_probability(pw_data_object.get_cgm_pw(), predicted_cgm)
        hyper_prob = Hypoglycemia_Hyperglycemia.get_hyperglycemia_probability(pw_data_object.get_cgm_pw(), predicted_cgm)
    else:
        hypo_prob = 0
        hyper_prob = 0

    pw_data_object.push_hypo_prob(hypo_prob)
    pw_data_object.push_hyper_prob(hyper_prob)

    # opt_insulin_trajectory is the actual values of insulin and not the deviation values
    opt_insulin = opt_insulin_trajectory[0]

    # basal_out unit is U/hr and bolus_out unit is U
    if opt_insulin <= 0:
        basal_out = 0
        bolus_out = 0
    elif (opt_insulin > 0) and (opt_insulin <= demo_info.get_basal_rate_mU_min()):
        basal_out = opt_insulin * 60.0 / 1000
        bolus_out = 0
    else:
        basal_out = demo_info.get_basal_rate()  # U/hr
        bolus_out = (opt_insulin - demo_info.get_basal_rate_mU_min()) / 1000 * T

    return basal_out, bolus_out

def mpc_set(linear_model: Linear_State_Space_Model, mpc_obj_fcn_params: MPC_Objective_Function_Params,
            PIC_constraints: Plasma_Insulin_Concentration_Constraint_Params, bounds: Bounds):

    """
        xss is the point around which the system is linearized
        xss is not necessarily steady state point, it could be any point

        order of decision variables
        decision variables = [{du}, y, e, {HUB, HLB, SUB, SLB}, {PICdev1, PICdev2}, {x}, {u}]

        number of decision variables
        number of decision variables = 10(states) + 4(u) + 4(du) + 1(y) + 1(e) + 2(PICdev) + 4(hard and soft bounds on PIC)

        the 5th decision variable at each sample time is y, thus the corresponding row in WeightMatrix is zero

        the 7th to 10th decision variables at each sample time are upper and lower values for PIC hard and soft constraint

        the 11th and 12th decision variables at each sample time are used to construct a soft constraint on PIC. if we call these decision
        variables PICdev1 and PICdev2 the soft PIC constraints are :
        PIC_LowerSoftBound - PICdev2 <= PIC <= PIC_UpperSoftBound + PICdev2
        in recent version PIC_LowerSoftBound = PIC_UpperSoftBound, thus we can also convert these linear inequality
        constraints to linear equality constraints
        hard coded number 6 in the following lines are actually the 6 decision variables that are defined at each
        sample time for incorporating PIC constraints
        these 6 decision variables are : [PIC_UpperHardBound, PIC_LowerHardBound, PIC_UpperSoftBound, PIC_LowerSoftBound, PICdev1, PICdev2]
    """

    xss = linear_model.x0
    yss = np.dot(linear_model.get_C().reshape(-1), xss)
    PIC_ss = xss[2]

    prediction_horizon = mpc_obj_fcn_params.prediction_horizon

    nx = linear_model.get_state_num()
    nu = linear_model.get_input_num_artificial()
    ny = linear_model.get_output_num()

    ineq_constraints_num = 4
    eq_constraints_num = nu+ny+ny+4+nx

    sample_time_decision_var_num = nu + ny + ny + 6 + nx + nu

    weight_matrix = np.zeros((sample_time_decision_var_num, sample_time_decision_var_num))
    weight_matrix[0:nu, 0:nu] = mpc_obj_fcn_params.Q_du  # du
    weight_matrix[nu+ny:nu+ny+ny, nu+ny:nu+ny+ny] = mpc_obj_fcn_params.Q_e  # e
    weight_matrix[nu+ny+ny+4:nu+ny+ny+6, nu+ny+ny+4:nu+ny+ny+6] = mpc_obj_fcn_params.Q_PIC  # PIC deviation
    weight_matrix[nu+ny+ny+6:nu+ny+ny+nx+6, nu+ny+ny+6:nu+ny+ny+nx+6] = mpc_obj_fcn_params.Q_x  # x (states)
    weight_matrix[nu+ny+ny+nx+6:nu+ny+ny+nx+nu+6, nu+ny+ny+nx+6:nu+ny+ny+nx+nu+6] = mpc_obj_fcn_params.Q_u  # u

    H = np.kron(np.eye(prediction_horizon), weight_matrix)
    f = np.zeros((sample_time_decision_var_num * prediction_horizon, 1))

    # inequality constraints
    A0_inequality_hard = np.array([
        np.concatenate([np.zeros(nu), np.zeros(2*ny), [-1, 0, 0, 0], np.zeros(2), Hovorka_Model_States_Object_1.get_PIC_pos(), np.zeros(nu)]),
        np.concatenate([np.zeros(nu), np.zeros(2*ny), [0, 1, 0, 0], np.zeros(2), -Hovorka_Model_States_Object_1.get_PIC_pos(), np.zeros(nu)])
    ])

    A0_inequality_soft = np.array([
        np.concatenate([np.zeros(nu), np.zeros(2*ny), [0, 0, -1, 0], [-1, 0], Hovorka_Model_States_Object_1.get_PIC_pos(), np.zeros(nu)]),
        np.concatenate([np.zeros(nu), np.zeros(2*ny), [0, 0, 0, 1], [0, -1], -Hovorka_Model_States_Object_1.get_PIC_pos(), np.zeros(nu)])
    ])

    A_inequality_hard = np.kron(np.eye(prediction_horizon), A0_inequality_hard)
    A_inequality_soft = np.kron(np.eye(prediction_horizon), A0_inequality_soft)

    if PIC_constraints.mode == 'all':
        A_inequality = np.concatenate([A_inequality_hard, A_inequality_soft])
    elif PIC_constraints.mode == 'hard':
        A_inequality = A_inequality_hard
    elif PIC_constraints.mode == 'soft':
        A_inequality = A_inequality_soft
    else:
        A_inequality = np.array([])

    b_inequality = np.zeros((A_inequality.shape[0], 1))

    # equality constraints
    A0_equality = np.concatenate([
        np.concatenate([np.zeros(nu), [1], np.zeros(ny+4+2+nx+nu)]).reshape(1, -1),  # y
        np.concatenate([np.zeros(nu), [1], [-1], np.zeros(4+2+nx+nu)]).reshape(1, -1),  # e
        np.concatenate([np.zeros(nu), [-PIC_constraints.get_upper_hard_slope()], [0], [1, 0, 0, 0], np.zeros(2+nx+nu)]).reshape(1, -1),  # PIC upper hard constraint
        np.concatenate([np.zeros(nu), [-PIC_constraints.get_lower_hard_slope()], [0], [0, 1, 0, 0], np.zeros(2+nx+nu)]).reshape(1, -1),  # PIC lower hard constraint
        np.concatenate([np.zeros(nu), [-PIC_constraints.get_soft_slope()], [0], [0, 0, 1, 0], np.zeros(2+nx+nu)]).reshape(1, -1),  # PIC soft constraint
        np.concatenate([np.zeros(nu), [-PIC_constraints.get_soft_slope()], [0], [0, 0, 0, 1], np.zeros(2+nx+nu)]).reshape(1, -1),  # PIC soft constraint
        np.concatenate([np.zeros((nx, nu+2*ny+4+2)), np.eye(nx), np.zeros((nx, nu))], axis=1),  # x
        np.concatenate([-np.eye(nu), np.zeros((nu, 2*ny+4+2+nx)), np.eye(nu)], axis=1)  # du
    ], axis=0)

    A0_equality = np.concatenate([A0_equality, np.zeros((A0_equality.shape[0], sample_time_decision_var_num * (prediction_horizon - 1)))], axis=1)

    A11_equality_block = np.concatenate([
        np.zeros((6, sample_time_decision_var_num)),
        np.concatenate([np.zeros((nx, nu+2*ny+4+2)), linear_model.get_A(), linear_model.get_B_artificial()], axis=1),
        np.concatenate([np.zeros((nu, sample_time_decision_var_num - nu)), -np.eye(nu)], axis=1)
        ], axis=0)

    A11_equality = np.kron(np.eye(prediction_horizon - 1), A11_equality_block)
    A11_equality = np.concatenate([A11_equality, np.zeros(((prediction_horizon - 1) * eq_constraints_num, sample_time_decision_var_num))], axis=1)

    A12_equality_block = np.concatenate([
            np.concatenate([np.zeros(nu), [1], np.zeros(ny+4+2), linear_model.get_C().reshape(-1), np.zeros(nu)]).reshape(1, -1),
            np.concatenate([np.zeros(nu), [1], [-1], np.zeros(4+2+nx+nu)]).reshape(1, -1),
            np.concatenate([np.zeros(nu), [-PIC_constraints.get_upper_hard_slope()], [0], [1, 0, 0, 0], np.zeros(2+nx+nu)]).reshape(1, -1),
            np.concatenate([np.zeros(nu), [-PIC_constraints.get_lower_hard_slope()], [0], [0, 1, 0, 0], np.zeros(2+nx+nu)]).reshape(1, -1),
            np.concatenate([np.zeros(nu), [-PIC_constraints.get_soft_slope()], [0], [0, 0, 1, 0], np.zeros(2+nx+nu)]).reshape(1, -1),
            np.concatenate([np.zeros(nu), [-PIC_constraints.get_soft_slope()], [0], [0, 0, 0, 1], np.zeros(2+nx+nu)]).reshape(1, -1),
            np.concatenate([np.zeros((nx, nu+2*ny+4+2)), -np.eye(nx), np.zeros((nx, nu))], axis=1),
            np.concatenate([-np.eye(nu), np.zeros((nu, 2*ny+4+2+nx)), np.eye(nu)], axis=1)
    ], axis=0)

    A12_equality = np.kron(np.eye(prediction_horizon - 1), A12_equality_block)
    A12_equality = np.concatenate([np.zeros(((prediction_horizon - 1) * eq_constraints_num, sample_time_decision_var_num)), A12_equality], axis=1)
    A1_equality = A11_equality + A12_equality
    A_equality = np.concatenate([A0_equality, A1_equality], axis=0)

    b0_equality = np.concatenate([
                            [0,  # y
                            mpc_obj_fcn_params.ysp,  # e
                            PIC_constraints.get_upper_hard_slope() * yss + PIC_constraints.get_upper_hard_intercept() - PIC_ss,  # PIC upper hard constraint
                            PIC_constraints.get_lower_hard_slope() * yss + PIC_constraints.get_lower_hard_intercept() - PIC_ss,  # PIC lower hard constraint
                            PIC_constraints.get_soft_slope() * yss + PIC_constraints.get_soft_intercept() - PIC_ss,  # PIC soft constraint
                            PIC_constraints.get_soft_slope() * yss + PIC_constraints.get_soft_intercept() - PIC_ss],  # PIC soft constraint
                            np.zeros(nx),  # x
                            np.zeros(nu)  # du
                            ]).reshape(-1, 1)

    b1_equality_block = np.concatenate([
                            [0,  # y
                            mpc_obj_fcn_params.ysp,  # e
                            PIC_constraints.get_upper_hard_slope() * yss + PIC_constraints.get_upper_hard_intercept() - PIC_ss,  # PIC upper hard constraint
                            PIC_constraints.get_lower_hard_slope() * yss + PIC_constraints.get_lower_hard_intercept() - PIC_ss,  # PIC lower hard constraint
                            PIC_constraints.get_soft_slope() * yss + PIC_constraints.get_soft_intercept() - PIC_ss,  # PIC soft constraint
                            PIC_constraints.get_soft_slope() * yss + PIC_constraints.get_soft_intercept() - PIC_ss],  # PIC soft constraint
                            np.zeros(nx),  # x
                            np.zeros(nu)  # du
                            ]).reshape(-1, 1)

    b1_equality = np.tile(b1_equality_block, (prediction_horizon - 1, 1))
    b_equality = np.concatenate([b0_equality, b1_equality], axis=0)

    first_input_indices = np.arange(sample_time_decision_var_num - nu, sample_time_decision_var_num * prediction_horizon, sample_time_decision_var_num)
    u_index = np.hstack([first_input_indices, first_input_indices + 1, first_input_indices + 2, first_input_indices + 3])
    u_index = np.sort(u_index)

    lb_ = np.concatenate([bounds.du_min, bounds.y_min, np.full(ny, -np.inf), np.full(6, -np.inf), bounds.x_min, np.zeros(nu)]).reshape(-1, 1)
    lb = np.tile(lb_, (prediction_horizon, 1))
    lb[u_index] = bounds.u_min.T.reshape(-1, 1)

    ub_ = np.concatenate([bounds.du_max, bounds.y_max, np.full(ny, np.inf), np.full(6, np.inf), bounds.x_max, np.zeros(nu)]).reshape(-1, 1)
    ub = np.tile(ub_, (prediction_horizon, 1))
    ub[u_index] = bounds.u_max.T.reshape(-1, 1)

    x0_ = np.concatenate([np.zeros(nu), [0], [0], np.ones(6), np.zeros(nx), np.zeros(nu)]).reshape(-1, 1)
    x0 = np.tile(x0_, (prediction_horizon, 1))

    qop = Quadratic_Optimization_Problem(H=H, f=f, A=A_inequality, b=b_inequality, Aeq=A_equality, beq=b_equality, lb=lb, ub=ub, x0=x0)

    return qop

def mpc_solve(qop: Quadratic_Optimization_Problem):
    # resolving the possible issue of P being positive semi-definite by adding a small number to the diagonal elements
    # eps = 1.0e-6
    # qop.H += eps * np.identity(qop.H.shape[0])

    try:
        # do not use 'cvxopt' or 'quadprog' as the solver
        # instead use 'clarabel' or 'highs' as the solver
        x_solution = qpsolvers.solve_qp(P=qop.H, q=qop.f, G=qop.A, h=qop.b, A=qop.Aeq, b=qop.beq, lb=qop.lb, ub=qop.ub, solver='clarabel')
        exit_flag = True
    except qpsolvers.exceptions.ProblemError:
        x_solution = None
        exit_flag = False

    return x_solution, exit_flag
