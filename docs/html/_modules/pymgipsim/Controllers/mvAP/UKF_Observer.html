def ukf_observer(x, z, U, noise_error_cov_mat: Noise_Error_Covariance_Matrix, hovorka_params: Hovorka_Parameters):
    # z is the measured output (z=Glucose concentration in mmol/L)
    L = len(x)
    L_X = 2 * L + 1
    alpha = 1
    ki = 0
    beta = 2

    P = noise_error_cov_mat.P_UKF
    Q = noise_error_cov_mat.Q_UKF
    R = noise_error_cov_mat.R_UKF

    lambda_ = alpha ** 2 * (L + ki) - L
    c = L + lambda_
    Wm = np.concatenate(([lambda_ / c], 0.5 / c * np.ones(2*L)))
    # Wm = np.array([lambda_ / c, 0.5 / c * np.ones((1, 2 * L))])

    Wc = np.copy(Wm)
    Wc[0] = Wc[0] + (1 - alpha ** 2 + beta)

    c = c ** 0.5

    A = c * cholesky_decomposition(P)
    Y = np.tile(x[:, None], (1, L))
    X = np.column_stack([x, Y + A, Y - A])

    X_priori = get_hovorka_model_ukf_observer(X, U, hovorka_params)
    X_priori = saturate_states(X_priori, hovorka_params)

    x_averaged = np.sum(Wm[None, :] * X_priori, axis=1)
    X_centered = X_priori - x_averaged[:, None]
    P_X = X_centered @ np.diag(Wc) @ X_centered.T + Q

    Y_priori = X_priori[3, :]
    Y_output = np.sum(Y_priori * Wm)

    Y_centered = Y_priori - Y_output
    P_Y = np.sum((Y_centered ** 2) * Wc) + R

    Pxx_yy = X_centered @ np.diag(Wc) @ Y_centered
    K = Pxx_yy / P_Y
    x = x_averaged + K * (z - Y_output)
    x = saturate_states(x[:, None], hovorka_params)
    P = P_X - K[:, None] @ Pxx_yy[None, :]

    noise_error_cov_mat.P_UKF = P

    return x.reshape(-1)

def saturate_states(X, hovorka_params: Hovorka_Parameters):
    num_row = X.shape[0]  # number of states
    num_col = X.shape[1]  # number of particles
    SIT = hovorka_params.SIT
    SID = hovorka_params.SID
    SIE = hovorka_params.SIE
    gain = hovorka_params.gain
    BW = hovorka_params.BW

    Upper_scale = 2
    Lower_scale = 3

    # Upper and lower bound values
    insulin_to_carb_ratio = 1 / 40  # unit per gram of carbs
    max_meal_size = 90  # grams
    Meal_max = max_meal_size / 5  # gr/min carbs   ==> the maximum amount of meal
    s1_min = np.finfo(float).eps
    s1_max = Meal_max * insulin_to_carb_ratio * 1000 * 65  # with 20 percent above threshold values
    s2_min = np.finfo(float).eps
    s2_max = s1_max

    ke_min = 0.175 / Lower_scale
    ke_max = 0.2 - np.finfo(float).eps

    I_min = 0
    I_max = 160

    G_min = (40 / 18)
    G_max = (400 / 18)

    Q2_min = 0
    Q2_max = 0.16 * BW * (G_max) / 0.066

    Tmaxe_max = 10

    EE_max = 20
    R1_min = 0
    R1_max = gain * EE_max * Tmaxe_max

    R2_min = 0
    R2_max = R1_max

    Tmaxi_min = 62.8 / Lower_scale
    Tmaxi_max = 62.8 * Upper_scale

    x1_min = 0
    x1_max = I_max * SIT

    x2_min = 0
    x2_max = I_max * SID

    x3_min = 0
    x3_max = I_max * SIE

    Ug_min = 0
    Ug_max = float('inf')

    lolim = np.array([s1_min, s2_min, I_min, G_min, Q2_min, R1_min, R2_min, x1_min, x2_min, x3_min, Tmaxi_min, ke_min, Ug_min])
    uplim = np.array([s1_max, s2_max, I_max, G_max, Q2_max, R1_max, R2_max, x1_max, x2_max, x3_max, Tmaxi_max, ke_max, Ug_max])

    LO = np.tile(lolim.reshape((num_row, 1)), num_col)
    UP = np.tile(uplim.reshape((num_row, 1)), num_col)

    lolimbreach = X < LO
    uplimbreach = X > UP

    X[lolimbreach] = LO[lolimbreach]
    X[uplimbreach] = UP[uplimbreach]

    return X
